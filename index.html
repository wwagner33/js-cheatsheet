JavaScript Cheat Sheet

Referência rápida da linguagem de programação JavaScript

Variáveis e Constantes

var nome = "João"; // Escopo de função
let idade = 30; // Escopo de bloco
const PI = 3.1416; // Constante, não pode ser reatribuída

Tipos de Dados

let texto = "String";
let numero = 42;
let booleano = true;
let nulo = null;
let indefinido = undefined;
let naoEhNumero = NaN;
let infinito = Infinity;
let simbolo = Symbol("desc");
let objetoLiteral = {chave: "valor"};

Operadores

// Aritméticos
+ - * / % **

// Comparação
== != === !== > < >= <=

// Lógicos
&& || !

// Atribuição
= += -= *= /=

Estruturas de Controle

// Condicional if
if (condicao) {
  // código
} else if (outraCondicao) {
  // código
} else {
  // código
}

// Switch
switch(valor) {
  case 1:
    // código
    break;
  default:
    // código
}

Loops

// For loop
for(let i = 0; i < 5; i++) {
  console.log(i);
}

// While loop
while (condicao) {
  // código
}

// For...of (arrays)
for(let item of array) {
  // código
}

Funções

// Declaração de função
function soma(a, b) {
  return a + b;
}

// Arrow function
const multiplicacao = (a, b) => a * b;

// Função anônima
const saudacao = function(nome) {
  return "Olá, " + nome;
};

Arrays

// Criando arrays
let frutas = ["Maçã", "Banana"];
let numeros = [1, 2, 3];

// Métodos comuns
frutas.push("Laranja"); // Adiciona ao final
frutas.pop(); // Remove do final
frutas.shift(); // Remove do início
frutas.unshift("Morango"); // Adiciona ao início
let copia = frutas.slice(); // Cria cópia

Objetos

// Criando objetos
let pessoa = {
  nome: "Maria",
  idade: 30,
  profissao: "Desenvolvedora",
  saudacao: function() {
    return "Olá, meu nome é " + this.nome;
  }
};

// Acessando propriedades
pessoa.nome; // "Maria"
pessoa["idade"]; // 30

Promises e Async/Await

// Promise
const minhaPromise = new Promise((resolve, reject) => {
  // código assíncrono
  if (sucesso) {
    resolve(resultado);
  } else {
    reject(erro);
  }
});

// Async/Await
async function fetchData() {
  try {
    const resultado = await minhaPromise;
    console.log(resultado);
  } catch (erro) {
    console.error(erro);
  }
}

DOM Manipulation

// Selecionando elementos
const elemento = document.getElementById("id");
const elementos = document.querySelectorAll(".classe");

// Alterando conteúdo
elemento.textContent = "Novo texto";
elemento.innerHTML = "<strong>HTML</strong>";

// Eventos
elemento.addEventListener("click", () => {
  console.log("Clicado!");
});

Métodos de Array

// Métodos úteis
array.map(item => item * 2); // Transforma cada item
array.filter(item => item > 5); // Filtra itens
array.reduce((acc, cur) => acc + cur, 0); // Acumula valores
array.find(item => item.id === 1); // Encontra primeiro item
array.some(item => item > 5); // Verifica se algum item satisfaz
array.every(item => item > 5); // Verifica se todos satisfazem

JSON

// Convertendo para JSON
const obj = {nome: "João", idade: 30};
const json = JSON.stringify(obj);
// '{"nome":"João","idade":30}'

// Convertendo de JSON
const objNovo = JSON.parse(json);
// {nome: "João", idade: 30}
